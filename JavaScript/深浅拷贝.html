function deepClone(newObj, oldObj) {
  for (let i in oldObj) {
    let item = oldObj[i]
    if (typeof item === 'function') {
      newObj[i] = item
    } else if (Array.isArray(item)) {
      newObj[i] = []
      deepClone(newObj, item)
    } else if (typeof item === 'object') {
      newObj[i] = {}
      deepClone(newObj, item)
    } else {
      newObj[i] = item
    }
  }
  return newObj
}
